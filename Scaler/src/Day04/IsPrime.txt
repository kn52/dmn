class Solution {
    public int solve(int A) {
        int _isprime = 1;
        if (A == 1) {
            return 0;
        }
        for (int i = 2; i <= A/2; i++) {
            if (A%i == 0) {
                _isprime = 0;
                break;
            }
        }
        return _isprime;
    }
}
