class Solution {
    public int solve(int A) {
        int count = 0;
        for (int i = 1; i <= Math.Sqrt(A); i++) {
            if (A%i == 0) {
                if (i*i == A)
                count += 1;
                else 
                count += 2;
            }
        }
        return count;
    }
}
